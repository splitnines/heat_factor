covid_dates = list(sorted(covid_data_df['dateRep'].unique()))
one_day = datetime.timedelta(days=1)

def autolabel(rects):
    """Attach a text label above each bar in *rects*, displaying its height."""
    for rect in rects:
        height = rect.get_height()
        ax.annotate('{}'.format(height),
        xy=(rect.get_x() + rect.get_width() / 2, height),
        xytext=(0, 0),
        textcoords="offset points",
        ha='center', va='bottom')

fig, ax = plt.subplots(figsize=(18, 8))

x = np.arange(len(list(top_10_by_deaths['df_usa']['cases'])))

count = 0
for country in top_10_by_deaths:
    if count < 6:
        ax.plot(x, (top_10_by_deaths[country]['cases'] / top_10_by_deaths[country]['popData2018']) * 100, label=top_10_by_deaths[country]['countriesAndTerritories'].iloc[0])
    else:
        ax.plot(x, (top_10_by_deaths[country]['cases'] / top_10_by_deaths[country]['popData2018']) * 100, label=top_10_by_deaths[country]['countriesAndTerritories'].iloc[0], linestyle='dotted')
    count += 1

ax.set(xlabel="Days", ylabel="Percent of Population Infected", title="COVID-19 Cases\n(Percent of Population)")

ax.set_xticklabels(np.arange(min(covid_dates), max(covid_dates) + one_day, step=3), rotation=90)
ax.set_xticks(np.arange(min(x), max(x) + 1, step=3))

plt.legend()
plt.style.use('seaborn')
plt.tight_layout()
plt.show()

ax.set_xticklabels(labels)
ax.set_xticks(x)
