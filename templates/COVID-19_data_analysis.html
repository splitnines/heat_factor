def autolabel(rects):
    """Attach a text label above each bar in *rects*, displaying its height."""
    for rect in rects:
        height = rect.get_height()
        ax.annotate('{}'.format(height),
        xy=(rect.get_x() + rect.get_width() / 2, height),
        xytext=(0, 0),
        textcoords="offset points",
        ha='center', va='bottom')
            
p_dict = dict(covid_data_df.groupby(['countriesAndTerritories'])['deaths'].sum().sort_values(ascending=False).iloc[0:10])
labels = [c for c in p_dict.keys()]
deaths = [d for d in p_dict.values()]
            
fig, ax = plt.subplots(figsize=(12, 6))

x = np.arange(len(labels))

rects = ax.bar(x, deaths, width=0.5)

autolabel(rects)

ax.set(xlabel="Country", ylabel="Total Deaths", title="COVID-19 Total Deaths")
ax.set_xticklabels(labels)
ax.set_xticks(x)

plt.style.use('seaborn')
plt.tight_layout()
plt.show()
